%%  - programa para resolver el factorial de un n√∫mero

program facto1;

var
    int inicial,numero,i;

main() {
    inicial = 1;

    write("Factorial de");
    
    read(numero);
    
    for i = 1 to numero do {
        inicial = inicial * i;
    }
    
    write(inicial);
}
